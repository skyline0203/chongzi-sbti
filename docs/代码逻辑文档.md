## 📌 读前须知

### 微信小程序的基本概念

微信小程序的代码分为四部分（每个页面都有这四个文件）：

| 文件后缀 | 类比 | 作用 |
|---------|------|------|
| `.js` | 大脑 | 逻辑：处理数据、响应操作 |
| `.wxml` | 骨架 | 界面：页面上有什么元素 |
| `.wxss` | 皮肤 | 样式：颜色、大小、间距 |
| `.json` | 配置 | 设置：导航栏标题等 |

### 和 C 语言的对比

```c
// C语言
int a = 10;           // 定义一个变量
if (a > 5) { }        // 条件判断
for (int i=0; i<10; i++) {}  // 循环
void func() {}        // 定义函数
```

```javascript
// 微信小程序 (JavaScript)
let a = 10;           // 定义一个变量（let = 可变, const = 不可变）
if (a > 5) { }        // 条件判断（一模一样！）
for (let i=0; i<10; i++) {}  // 循环（几乎一样！）
function func() {}    // 定义函数
```

**好消息**：JavaScript 的语法和 C 语言非常像，你会 C 就能读懂大部分 JS。

### 1. app.js — 小程序总开关

```javascript
App({                          // App() = 创建一个小程序实例（所有页面共享）
  onLaunch() {                 // onLaunch = 小程序启动时自动执行（类似 main 函数）

  },
  globalData: {                // 全局数据 = 所有页面都能访问的变量

    cdnBase: ''                // CDN路径（预留，暂未使用）

```

**类比 C 语言**：
- `App()` ≈ `int main()`
- `onLaunch()` ≈ 程序启动时最先执行的那个函数
- `globalData` ≈ 全局变量

### 4. pages/quiz/quiz.js — 答题页逻辑（核心！）

这是整个项目最重要的文件。

```javascript
// require = 导入另一个文件的东西（类似 C 的 #include）
const { questions } = require('../../utils/questions.js');

Page({
  data: {
    questions: [],              // 所有题目（从 questions.js 导入）
    currentIndex: 0,            // 当前是第几题（从 0 开始计数）
    total: 15,                  // 总题数
    currentQuestion: {},        // 当前显示的那道题
    selectedOption: -1,         // 当前选中了第几个选项（-1 = 没选）
    answers: [],                // 所有题目的答案（数组）
    progress: 0,                // 进度条百分比（0-100）
    optionLetters: ['A','B','C','D']  // 选项的字母标签
  },

  // ========== 页面加载 ==========
  onLoad() {
    this.setData({              // setData = 修改 data 中的值，页面会自动刷新
      questions: questions,     // 把导入的题目赋值给页面数据
      currentQuestion: questions[0],  // 第一题作为当前题
      total: questions.length,  // 题目总数
      progress: (1 / questions.length) * 100  // 进度 = 当前题/总题数 × 100
    });
  },

  // ========== 选择选项 ==========
  // e = 事件对象，包含了用户点击了哪个元素
  selectOption(e) {
    // dataset.index = 从 wxml 的 data-index 属性获取的值
    const index = e.currentTarget.dataset.index;
    
    // 从当前题目中获取被选中选项的 type（S/H/I/T）
    const type = this.data.currentQuestion.options[index].type;

    // 复制一份答案数组（用 ... 展开运算符，避免直接修改原数组）
    const answers = [...this.data.answers];
    // 把当前题的答案存进去
    answers[this.data.currentIndex] = type;

    // 更新页面数据
    this.setData({
      selectedOption: index,    // 标记选中了第几个选项
      answers: answers          // 更新答案数组
    });
  },

  // ========== 下一题 ==========
  nextQuestion() {
    // 先检查：有没有选答案？
    if (this.data.selectedOption === -1) {
      wx.showToast({ title: '先选一个吧！', icon: 'none' });  // 弹出提示
      return;  // 没选就直接返回，不往下走
    }

    // 检查：是不是最后一题？
    if (this.data.currentIndex === this.data.total - 1) {
      this.calculateResult();  // 最后一题 → 计算结果
      return;
    }

    // 不是最后一题 → 前进到下一题
    const nextIndex = this.data.currentIndex + 1;
    const answers = this.data.answers;

    this.setData({
      currentIndex: nextIndex,
      currentQuestion: this.data.questions[nextIndex],
      // 如果下一题之前已经选过答案，恢复选中状态
      selectedOption: answers[nextIndex] !== undefined ?
        this.getTypeIndex(this.data.questions[nextIndex], answers[nextIndex]) : -1,
      progress: ((nextIndex + 1) / this.data.total) * 100
    });
  },

  // ========== 上一题 ==========
  prevQuestion() {
    if (this.data.currentIndex === 0) return;  // 第一题，没法往前了

    const prevIndex = this.data.currentIndex - 1;
    const answers = this.data.answers;

    this.setData({
      currentIndex: prevIndex,
      currentQuestion: this.data.questions[prevIndex],
      selectedOption: answers[prevIndex] !== undefined ?
        this.getTypeIndex(this.data.questions[prevIndex], answers[prevIndex]) : -1,
      progress: ((prevIndex + 1) / this.data.total) * 100
    });
  },

  // ========== 辅助函数：找选项的下标 ==========
  // 给定一道题和 type，返回该 type 在选项中的下标（0/1/2/3）
  getTypeIndex(question, type) {
    for (let i = 0; i < question.options.length; i++) {
      if (question.options[i].type === type) return i;
    }
    return -1;  // 没找到
  },

  // ========== 计算最终结果 ==========
  calculateResult() {
    // 统计每种类型各选了几次
    const counts = { S: 0, H: 0, I: 0, T: 0 };
    this.data.answers.forEach(type => {
      if (counts[type] !== undefined) counts[type]++;
    });

    // 找出选中次数最多的类型
    let maxType = 'T';
    let maxCount = 0;
    for (const type in counts) {
      if (counts[type] > maxCount) {
        maxCount = counts[type];
        maxType = type;
      }
    }

    // 跳转到结果页，把类型和统计数字传过去
    wx.redirectTo({                          // redirectTo = 替换当前页面（不能返回）
      url: `/pages/result/result?type=${maxType}&counts=${JSON.stringify(counts)}`
    });
  }
})
```

**算法总结**（用 C 的伪代码表示）：
```
统计 answers[] 数组中 S/H/I/T 各出现几次
找出出现次数最多的那个
那就是你的鉴定结果
```

## 🔐 登录模块逻辑（新增）

### pages/login/login.js

```javascript
Page({
  data: {
    isAdmin: false,       // 是否管理员登录模式
    isLoggedIn: false,    // 是否已登录
    userInfo: null,       // 用户信息
    loading: false        // 是否正在加载
  },

  onLoad(options) {
    // 从 URL 参数判断是普通登录还是管理员登录
    this.setData({
      isAdmin: options.role === 'admin'
    });
  },

  // ========== 普通用户微信登录 ==========
  async userLogin() {
    this.setData({ loading: true });
    
    try {
      // 第一步：获取微信登录凭证（code）
      // wx.login() 是微信提供的 API，不需要用户确认
      const loginRes = await wx.p.login();
      
      // 第二步：获取用户信息（需要用户点击授权按钮）
      const profileRes = await wx.p.getUserProfile({
        desc: '用于完善用户资料'
      });
      
      // 第三步：把 code 发送到服务端换取 openid
      // openid 是用户在小程序中的唯一标识
      const openid = await this.codeToOpenid(loginRes.code);
      
      // 第四步：保存登录状态到本地
      wx.setStorageSync('userInfo', profileRes.userInfo);
      wx.setStorageSync('openid', openid);
      wx.setStorageSync('isLoggedIn', true);
      
      this.setData({
        isLoggedIn: true,
        userInfo: profileRes.userInfo,
        loading: false
      });
      
      // 登录成功后跳转首页
      wx.switchTab({ url: '/pages/index/index' });
      
    } catch (err) {
      this.setData({ loading: false });
      wx.showToast({ title: '登录失败，请重试', icon: 'none' });
    }
  },

  // ========== 管理员微信登录 ==========
  async adminLogin() {
    this.setData({ loading: true });
    
    try {
      const loginRes = await wx.p.login();
      const openid = await this.codeToOpenid(loginRes.code);
      
      // 验证是否是管理员（服务端检查白名单）
      const isAdmin = await this.verifyAdmin(openid);
      
      if (isAdmin) {
        wx.setStorageSync('isAdmin', true);
        wx.setStorageSync('adminOpenid', openid);
        wx.redirectTo({ url: '/pages/admin/admin' });
      } else {
        wx.showToast({ title: '你不是管理员', icon: 'none' });
      }
    } catch (err) {
      wx.showToast({ title: '验证失败', icon: 'none' });
    }
    
    this.setData({ loading: false });
  },

  // ========== 工具函数 ==========
  
  // 把微信登录凭证发到服务端换取 openid
  async codeToOpenid(code) {
    // wx.request = 发送网络请求（类似 C 的 socket/http 请求）
    return new Promise((resolve, reject) => {
      wx.request({
        url: 'https://你的服务器/api/login',  // 替换成真实地址
        method: 'POST',
        data: { code: code },
        success: (res) => resolve(res.data.openid),
        fail: (err) => reject(err)
      });
    });
  },

  // 验证管理员身份
  async verifyAdmin(openid) {
    return new Promise((resolve, reject) => {
      wx.request({
        url: 'https://你的服务器/api/admin/verify',
        method: 'POST',
        data: { openid: openid },
        success: (res) => resolve(res.data.isAdmin),
        fail: (err) => reject(err)
      });
    });
  }
})
```

## 🔄 数据流向图

```
[首页] → 点击"开始鉴定" → 检查登录状态
                                ↓
                          未登录？→ [登录页] → 微信授权 → 返回首页
                                ↓
                          已登录？→ [答题页]
                                ↓
                          选择答案 → 存入 answers[] 数组
                                ↓
                          答完15题 → calculateResult()
                                ↓
                          统计 S/H/I/T 各几次 → 找出最多的
                                ↓
                          [结果页] → 显示结果 → 可保存/分享
```

## 💡 编程概念速查表

| 概念 | JavaScript | C 语言对比 | 说明 |
|------|-----------|-----------|------|
| 变量 | `let a = 1` | `int a = 1` | let 可变，const 不可变 |
| 常量 | `const a = 1` | `const int a = 1` | 不可修改 |
| 数组 | `[1, 2, 3]` | `int arr[3] = {1,2,3}` | 有序列表 |
| 对象 | `{key: value}` | struct | 键值对 |
| 函数 | `function f() {}` | `void f() {}` | 几乎一样 |
| 箭头函数 | `() => {}` | — | JS 特有的简写 |
| 导入 | `require('...')` | `#include` | 引入其他文件 |
| 导出 | `module.exports` | `extern` | 让别人用我的东西 |
| 异步 | `async/await` | — | JS 特有的，处理网络请求 |
| 回调 | `success: (res) => {}` | 函数指针 | 传一个函数给另一个函数 |
